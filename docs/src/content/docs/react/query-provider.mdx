**Overview of the QueryProvider**

All projects must be wrapped in the QueryProvider. This provides context to `@tanstack/react-query`

This is how your `pages/_app.tsx` should look like:

```tsx
import { type AppType } from 'next/dist/shared/lib/utils'
import '~/styles/globals.css'

import { QueryProvider } from '@prpc/react'
import { QueryClient } from '@tanstack/react-query'

const queryClient = new QueryClient()

const MyApp: AppType = ({ Component, pageProps }) => {
  return (
    <QueryProvider queryClient={queryClient}>
      <Component {...pageProps} />
    </QueryProvider>
  )
}

export default MyApp
```
